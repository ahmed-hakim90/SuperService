    <script>
        // محاكاة للوظائف الأساسية
        document.addEventListener('DOMContentLoaded', function() {
            // عناصر الصفحات
            const loginPage = document.getElementById('login-page');
            const signupPage = document.getElementById('signup-page');
            const waitingPage = document.getElementById('waiting-page');
            const dashboard = document.getElementById('dashboard');
            
            // عناصر النماذج
            const loginForm = document.getElementById('login-form');
            const signupForm = document.getElementById('signup-form');
            
            // روابط التنقل
            const showSignupLink = document.getElementById('show-signup');
            const showLoginLink = document.getElementById('show-login');
            const backToLoginLink = document.getElementById('back-to-login');
            const logoutLink = document.getElementById('logout');
            
            // التنقل بين الصفحات في لوحة التحكم
            const navLinks = document.querySelectorAll('.sidebar a.nav-link');
            const pageContents = document.querySelectorAll('.page-content');
            
            // عرض صفحة التسجيل
            showSignupLink.addEventListener('click', function(e) {
                e.preventDefault();
                loginPage.style.display = 'none';
                signupPage.style.display = 'block';
            });
            
            // عرض صفحة تسجيل الدخول
            showLoginLink.addEventListener('click', function(e) {
                e.preventDefault();
                signupPage.style.display = 'none';
                waitingPage.style.display = 'none';
                loginPage.style.display = 'block';
            });
            
            // العودة لتسجيل الدخول من صفحة الانتظار
            backToLoginLink.addEventListener('click', function(e) {
                e.preventDefault();
                waitingPage.style.display = 'none';
                loginPage.style.display = 'block';
            });
            
            // معالجة تسجيل الدخول
            loginForm.addEventListener('submit', function(e) {
                e.preventDefault();
                // هنا سيتم الاتصال بـ Supabase للمصادقة
                loginPage.style.display = 'none';
                dashboard.style.display = 'block';
            });
            
            // معالجة إنشاء حساب
            signupForm.addEventListener('submit', function(e) {
                e.preventDefault();
                // هنا سيتم الاتصال بـ Supabase لإنشاء حساب
                signupPage.style.display = 'none';
                waitingPage.style.display = 'block';
            });
            
            // تسجيل الخروج
            logoutLink.addEventListener('click', function(e) {
                e.preventDefault();
                // هنا سيتم الاتصال بـ Supabase لتسجيل الخروج
                dashboard.style.display = 'none';
                loginPage.style.display = 'block';
            });
            
            // التنقل بين الصفحات في لوحة التحكم
            navLinks.forEach(link => {
                link.addEventListener('click', function(e) {
                    e.preventDefault();
                    
                    // إزالة النشاط من جميع الروابط
                    navLinks.forEach(l => l.classList.remove('active'));
                    
                    // إضافة النشاط للرابط الحالي
                    this.classList.add('active');
                    
                    // إخفاء جميع محتويات الصفحات
                    pageContents.forEach(page => page.classList.remove('active'));
                    
                    // عرض المحتوى المطلوب
                    const pageId = this.getAttribute('data-page');
                    document.getElementById(pageId).classList.add('active');
                });
            });
        });
    </script>
